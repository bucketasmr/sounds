// === БЕЗОПАСНЫЙ СЧЕТЧИК КЛИКОВ ДЛЯ IPHONE И FIREBASE === function trackLinkClick(platform, url) { const newWindow = window.open(url, '_blank'); try { if (typeof firebase !== 'undefined' && firebase.database) { const clickRef = firebase.database().ref('link_clicks/' + platform); clickRef.transaction((currentValue) => { return (currentValue || 0) + 1; }); } } catch (error) { console.error("Ошибка трекинга ссылок: " + error.message); } } // === ЛОГИКА ЭКРАНА "CONTINUE" ПРИ РАЗБЛОКИРОВКЕ ДЛЯ IOS === // 1. Создаем полупрозрачный экран блокировки const wakeUpOverlay = document.createElement('div'); wakeUpOverlay.style.cssText = ` position: fixed; top: 0; left: 0; width: 100%; height: 100%; background: rgba(5, 5, 5, 0.85); backdrop-filter: blur(8px); -webkit-backdrop-filter: blur(8px); z-index: 100000; display: none; flex-direction: column; align-items: center; justify-content: center; transition: opacity 0.3s ease; opacity: 0; `; // 2. Создаем кнопку Continue const continueBtn = document.createElement('button'); continueBtn.innerText = "Continue"; continueBtn.style.cssText = ` padding: 14px 40px; font-size: 18px; font-weight: 600; color: white; background: linear-gradient(135deg, #8a2be2 0%, #4a00e0 100%); border: none; border-radius: 30px; cursor: pointer; box-shadow: 0 4px 15px rgba(138, 43, 226, 0.4); transition: transform 0.2s ease, box-shadow 0.2s ease; -webkit-tap-highlight-color: transparent; `; wakeUpOverlay.appendChild(continueBtn); document.body.appendChild(wakeUpOverlay); // Показ экрана function showWakeUpScreen() { const isIOS = /iPad|iPhone|iPod/.test(navigator.userAgent) || (navigator.platform === 'MacIntel' && navigator.maxTouchPoints > 1); if (isIOS) { wakeUpOverlay.style.display = 'flex'; setTimeout(() => { wakeUpOverlay.style.opacity = '1'; }, 10); } } // Клик по кнопке Continue: Принудительное пересоздание записи в БД function handleContinueClick() { try { if (typeof firebase !== 'undefined' && firebase.database) { const database = firebase.database(); // Очищаем старый ключ сессии из памяти телефона, чтобы сгенерировать полностью новый sessionStorage.removeItem('firebase_online_key'); // Создаем абсолютно новую уникальную запись в узле online_users const newOnlineRef = database.ref('online_users').push(); // Сохраняем новый ключ в sessionStorage sessionStorage.setItem('firebase_online_key', newOnlineRef.key); // Записываем данные на сервер newOnlineRef.set({ status: "online", last_seen: firebase.database.ServerValue.TIMESTAMP }).then(() => { console.log("Новая сессия успешно создана, онлайн обновлен!"); }); // Настраиваем автоудаление при следующем реальном выходе newOnlineRef.onDisconnect().remove(); } } catch (e) { console.error("Ошибка принудительного обновления сессии:", e); } // Плавно убираем экран wakeUpOverlay.style.opacity = '0'; setTimeout(() => { wakeUpOverlay.style.display = 'none'; }, 300); } // Слушатели событий continueBtn.addEventListener('click', handleContinueClick); continueBtn.addEventListener('touchstart', () => { continueBtn.style.transform = 'scale(0.95)'; }); continueBtn.addEventListener('touchend', () => { continueBtn.style.transform = 'scale(1)'; }); document.addEventListener('visibilitychange', () => { if (!document.hidden) showWakeUpScreen(); }); window.addEventListener('pageshow', (event) => { if (event.persisted) showWakeUpScreen(); }); // Логика для выпадающего списка соцсетей (Links) const linksTrigger = document.getElementById('linksTrigger'); const linksOptions = document.getElementById('linksOptions'); const speedDropdown = document.getElementById('speedDropdown'); if (linksTrigger && linksOptions) { linksTrigger.addEventListener('click', (e) => { e.stopPropagation(); if (speedDropdown) speedDropdown.classList.remove('open'); linksOptions.classList.toggle('open'); }); window.addEventListener('click', () => { linksOptions.classList.remove('open'); }); }
